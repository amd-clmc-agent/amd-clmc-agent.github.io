280  void get_data_indexes(){
281  // create data stream names.
282  std::vector<std::string>
283  gyroscope_names (3),
284  accelerometer_names (3),
285  unfiltered_pos_names (Robot::n_dofs_),
286  unfiltered_vel_names (Robot::n_dofs_),
287  filtered_pos_names (Robot::n_dofs_),
288  filtered_vel_names (Robot::n_dofs_),
289  vicon_base_pose_names(16);
290  // root name of all the data
291  std::string task_name = "AthenaTestTask_";
292  std::string estimator_name = "vicon_IMU_BaseEstimator_";
293  // get the pos and vel names
294  std::vector<std::string> joint_names = JOINT_NAMES();
295  for(int i=0; i<Robot::n_dofs_; ++i){
296  unfiltered_pos_names[i] = task_name + "jraw_" + joint_names[i+1];
297  unfiltered_vel_names[i] = task_name + "djraw_" + joint_names[i+1];
298  filtered_pos_names[i] = task_name + "jfilt_" + joint_names[i+1];
299  filtered_vel_names[i] = task_name + "djfilt_" + joint_names[i+1];
300  }
301  // get the imu related names
302  gyroscope_names[0] = task_name + "imu_base_gyro_x";
303  gyroscope_names[1] = task_name + "imu_base_gyro_y";
304  gyroscope_names[2] = task_name + "imu_base_gyro_z";
305  accelerometer_names[0] = task_name + "imu_base_acc_x";
306  accelerometer_names[1] = task_name + "imu_base_acc_y";
307  accelerometer_names[2] = task_name + "imu_base_acc_z";
308  // get the name of the vicon hip frame
309  for(unsigned i=0 ; i<4 ; ++i){
310  for(unsigned j=0 ; j<4 ; ++j){
311  std::ostringstream tmp;
312  tmp << "vicon_hip_frame_" << i+1 << "_" << j+1 ;
313  vicon_base_pose_names[4*i + j] = tmp.str();
314  }
315  }
316  // get the ids of the data.
317  data_reader_.getIndexes(gyroscope_names, gyroscope_ids_);
318  data_reader_.getIndexes(accelerometer_names, accelerometer_ids_);
319  data_reader_.getIndexes(vicon_base_pose_names, vicon_base_pose_ids_);
320  vicon_is_new_frame_id_ = data_reader_.getIndex(estimator_name +
321  "is_new_frame");
322  }
