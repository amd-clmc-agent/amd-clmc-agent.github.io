79 inline Eigen::Matrix3d OrientationFilter::accelToRot(const Eigen::Vector3d &accel) {
80  // calculate roll and pitch angles from accelerometer and form orientation matrix
81  // accel measures acceleration (ie z=-9.81) and returns orientation of IMU wrt global frame.
82 
83  Eigen::Matrix3d R;
84 
85  double p = atan2(-accel(1), -accel(2));
86  double r = atan2(accel(0), -accel(1)*sin(p)-accel(2)*cos(p));
87  double y = 0.0;
88 
89  double c1 = cos(p);
90  double c2 = cos(r);
91  double c3 = cos(y);
92 
93  double s1 = sin(p);
94  double s2 = sin(r);
95  double s3 = sin(y);
96 
97  R(0,0) = c2*c3;
98  R(0,1) = c3*s1*s2 - c1*s3;
99  R(0,2) = c1*s2*c3 + s1*s3;
100 
101  R(1,0) = c2*s3;
102  R(1,1) = s1*s2*s3 + c1*c3;
103  R(1,2) = c1*s2*s3 - s1*c3;
104 
105  R(2,0) = -s2;
106  R(2,1) = s1*c2;
107  R(2,2) = c1*c2;
108 
109  return R;
110 }
