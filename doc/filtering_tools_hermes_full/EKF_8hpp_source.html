52 void EKF<S, M>::updateFilter(bool update_on) {
53 
54  // set to true if you want the check activated (!Non RT safe)
55  // check_jacobians(false);
56 
57  // compute the jacobians
58  if(numerical_jac_) {
59  computeJacobians(proc_jac_, meas_jac_, 1e-12);
60  } else {
61  formProcessJacobian(); // proc_jac_
62  formMeasJacobian(); // meas_jac_
63  }
64 
65  // compute the covariances
66  formProcessNoise(); // proc_noise_
67  formMeasNoise(); // meas_noise_
68 
69  // get the actual measurement data
70  formActualMeas(); // meas_actual_
71 
72  // discretize process jacobian and noise matrix using Van Loan's method
73  // (http://wolfweb.unr.edu/~fadali/EE782/NumEvalQphi.pdf):
74  if(is_discrete_) {
75  proc_jac_disc_ = proc_jac_;
76  proc_noise_disc_ = proc_noise_;
77  } else {
78  if(lin_order_ == 2){// order 2 linearization of the process dynamics
79  // in the continuous time system case we compute noise_jac_
80  formNoiseJacobian();// noise_jac_
81  VL_mat_.block(0,0,N,N) = - proc_jac_;
82  VL_mat_.block(0,N,N,N) = noise_jac_ * proc_noise_ * noise_jac_.transpose();
83  VL_mat_.block(N,N,N,N) = proc_jac_.transpose();
84  exp_VL_mat_ = Eigen::MatrixXd::Identity(2 * N, 2 * N) + dt_*VL_mat_ +
85  0.5 * dt_ * dt_ * VL_mat_ * VL_mat_;
86  proc_jac_disc_ = exp_VL_mat_.block(N,N,N,N).transpose();
87  proc_noise_disc_ = proc_jac_disc_ * exp_VL_mat_.block(0,N,N,N);
88 
89  }else{// order 1 linearization of the process dynamic
90  proc_jac_disc_ = Eigen::MatrixXd::Identity(N, N) + dt_ * proc_jac_;
91  proc_noise_disc_ = proc_noise_ / dt_;
92  }
93  }
94 
95  // assume measurements are discrete, else need to divide by dt:
96  // meas_noise_disc_ = meas_noise_/dt_;
97  meas_noise_disc_ = meas_noise_;
98 
99  // predict:
100  state_pre_ = processModel(state_post_);
101  state_pre_.state_cov =
102  proc_jac_disc_ * state_post_.state_cov * proc_jac_disc_.transpose() +
103  proc_noise_disc_;
104 
105  // update:
106  if(update_on) {
107 
108  meas_pred_ = measModel(state_pre_);
109  meas_pred_.meas_cov =
110  meas_jac_ * state_pre_.state_cov * meas_jac_.transpose() +
111  meas_noise_disc_;
112 
113  ldlt.compute(meas_pred_.meas_cov);
114  gain_mat_ = (ldlt.solve(meas_jac_)).transpose();
115  gain_mat_ = state_pre_.state_cov * gain_mat_;
116 
117  state_post_ = state_pre_.getState() + gain_mat_ * (meas_actual_-meas_pred_);
118  state_post_.state_cov = state_pre_.state_cov -
119  gain_mat_ * meas_jac_ * state_pre_.state_cov;
120 
121  } else {
122 
123  state_post_ = state_pre_;
124  state_post_.state_cov = state_pre_.state_cov;
125 
126  }
127 }

129 template <class S, class M>
130 void EKF<S, M>::computeJacobians(Eigen::MatrixXd &G_t,
131  Eigen::MatrixXd &H_t, double ds) {
132  // reset the matrices
133  G_t.setZero();
134  H_t.setZero();
135  // Compute G_t:
136  Eigen::Matrix<double,N,1> dx, diff_x;
137  S x_m, x_p, x_mdot, x_pdot;
138  // Compute H_t:
139  M s_m, s_p;
140  Eigen::Matrix<double,K,1> diff_s;
141 
142  for(int i=0; i<N; ++i) {
143 
144  dx.setZero();
145 
146  // minus:
147  dx(i) = -ds;
148 
149  x_m = state_post_ + dx;
150 
151  x_mdot = processModel(x_m);
152  s_m = measModel(x_m);
153 
154  // plus:
155  dx(i) = ds;
156 
157  x_p = state_post_ + dx;
158 
159  x_pdot = processModel(x_p);
160  s_p = measModel(x_p);
161 
162  // central finite difference:
163  diff_x = x_pdot - x_mdot;
164  diff_s = s_p - s_m;
165 
166  G_t.block(0,i,N,1) = diff_x/(2*ds);
167  H_t.block(0,i,K,1) = diff_s/(2*ds);
168 
169  }
170 }

172 template <class S, class M>
173 bool EKF<S, M>::check_jacobians(bool evaluate)
174 {
175  bool ok = true;
176  if (evaluate){
177  computeJacobians(proc_jac_, meas_jac_, 1e-12);
178  Eigen::MatrixXd proc_jac_fd = proc_jac_;
179  Eigen::MatrixXd meas_jac_fd = meas_jac_;
180  formProcessJacobian(); // proc_jac_
181  formMeasJacobian(); // meas_jac_
182  double error_proc_jac =
183  (proc_jac_fd -
184  Eigen::MatrixXd::Identity(N, N) - dt_ * proc_jac_).norm();
185  double meas_proc_jac = (meas_jac_fd - meas_jac_).norm();
186  assert(error_proc_jac < 1e-3 && "The process jacobian is wrong");
187  assert(meas_proc_jac < 1e-3 && "The measure jacobian is wrong");
188  ok &= error_proc_jac < 1e-3;
189  ok &= meas_proc_jac < 1e-3;
190  }
191  return ok;
192 }
