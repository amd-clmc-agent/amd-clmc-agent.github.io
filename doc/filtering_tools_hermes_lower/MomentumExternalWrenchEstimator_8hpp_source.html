77  // Use the most recent force/torque readings available:
78  Eigen::Matrix<double, 6, 1> contact_wrench_world;
79  com_contact_wrench_.setZero();
80  for(int i = 0; i < Robot::n_endeffs_; ++i) {
81  contact_wrench_world = contact_helper_->getMeasuredForceAtEndeffWorld(i + 1);
82  com_contact_wrench_.segment(0, 3) += contact_wrench_world.segment(0, 3);
83  com_contact_wrench_.segment(3, 3) += geometry_utils::Transformations::vector_to_skew_matrix(
84  kinematics_->endeffector_position(i + 1) - s.com_pos) * contact_wrench_world.segment(0, 3) +
85  contact_wrench_world.segment(3, 3);
86  }
87 
88  Eigen::Matrix<double, MomentumExternalWrenchEstimatorState::state_dim, 1> out;
89  out.segment(0, 3) = s.com_pos + dt_ * ((1.0 / kinematics_->robot_mass()) * s.lin_mom);
90  out.segment(3, 3) = s.lin_mom + dt_ * (com_contact_wrench_.segment(0, 3) +
91  kinematics_->robot_mass() * grav_vec_ + s.force_bias);
92  out.segment(6, 3) = s.ang_mom + dt_ * (com_contact_wrench_.segment(3, 3) + s.torque_bias);
93  out.segment(9, 3) = s.force_bias;
94  out.segment(12, 3) = s.torque_bias;
