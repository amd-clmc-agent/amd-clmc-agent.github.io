117 ViconBaseStateEstimator::processModel(ViconBaseStateEstimatorState &s) {
118  Eigen::Matrix<double, ViconBaseStateEstimatorState::state_dim, 1> out;
119 
120  // Predict the filter state:
121  Eigen::Matrix3d C = s.base_quat.quaternion_to_rotation_matrix();
122  out.segment(0, 3) = s.base_pos + dt_ * s.base_vel;
123  out.segment(3, 3) = s.base_vel + dt_ * (C.transpose() * (imu_accel_ - s.accel_bias) + grav_vec_);
124  out.segment(6, 3) = (s.base_quat + dt_ * (imu_angvel_ - s.gyro_bias)).log();
125  out.segment(9, 3) = s.accel_bias;
126  out.segment(12, 3) = s.gyro_bias;
127 
128  return out;
129 }
