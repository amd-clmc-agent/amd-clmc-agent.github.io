94  template<typename errVecDer>
95  void cs_computeOrientationError(const Eigen::Quaterniond& des_rot,
96  const Eigen::Quaterniond& cur_rot, Eigen::MatrixBase<errVecDer>& err_vec)
97  {
98  Eigen::Quaterniond q1 = des_rot;
99  Eigen::Quaterniond q2 = cur_rot;
100 
101  q1.normalize();
102  q2.normalize();
103 
104  // fix sign
105  if(q1.dot(q2)<0.0)
106  {
107  q1.vec() = -q1.vec();
108  q1.w() = -q1.w();
109  }
110 
111  Eigen::Quaterniond delta_q = q1 * q2.inverse();
112 
113  //make sure the q0 > 0 (to make the sine approx correct)
114  if(delta_q.w()<0)
115  {
116  delta_q.w() = -delta_q.w();
117  delta_q.x() = -delta_q.x();
118  delta_q.y() = -delta_q.y();
119  delta_q.z() = -delta_q.z();
120  }
121 
122  //use a linear approximation
123  if(fabs(1.0-delta_q.w()) < 0.01)
124  {
125  err_vec = 2.0 * delta_q.vec();
126  }
127  else //or the sine formulae
128  {
129  double alpha_norm = 2.0 * acos(delta_q.w());
130  err_vec = alpha_norm/sin(alpha_norm/2) * delta_q.vec();
131  }
132  }

142 
143  // fix sign
144  if(orient_meas_.dot(orient_des_)<0.0)
145  {
146  orient_meas_.vec() = -orient_meas_.vec();
147  orient_meas_.w() = -orient_meas_.w();
148  }
149  Eigen::Vector3d orient_err;

151  cs_computeOrientationError(orient_des_, orient_meas_, orient_err);

153  pos_orient_err.head<3>() = pos_des_-pos;
