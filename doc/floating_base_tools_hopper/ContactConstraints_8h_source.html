138 void ContactConstraints< Max_Ineq_Rows, Max_Eq_Rows>::updateAfterSolutionFound(){
139  adms_cop_ = ContactHelper::center_of_pressure(
140  opt_prob_composer_->getAdmisForceAtEndeff(endeff_id_).template head<3>(),
141  opt_prob_composer_->getAdmisForceAtEndeff(endeff_id_).template tail<3>(),
142  kinematics_->endeffector_position(endeff_id_),
143  endeffector_pose(), adms_cop_torque_);
144  double f_z = opt_prob_composer_->getAdmisForceAtEndeff(endeff_id_)[2];
145  if(std::abs(f_z) < 1e-4){
146  adms_cop_ *= 0.;
147  adms_cop_torque_ *= 0.;
148  }
149  else{
150  adms_cop_torque_ /= f_z;
151  }
152  cop_slack_ = cop_weight_inv_*cop_mat_*opt_prob_composer_->getAdmisForceAtEndeff(endeff_id_);
153  opt_prob_composer_->inequalitySlack(cop_slack_);
154  cop_slack_sqr_norm_ = cop_slack_.squaredNorm();
155 
156  fric_slack_ = fric_weight_inv_*fric_mat_*opt_prob_composer_->getAdmisForceAtEndeff(endeff_id_);
157  opt_prob_composer_->inequalitySlack(fric_slack_);

159  torsion_slack_ = torsion_weight_inv_*torsion_mat_*opt_prob_composer_->getAdmisForceAtEndeff(endeff_id_);
160  opt_prob_composer_->inequalitySlack(torsion_slack_);
161 }
