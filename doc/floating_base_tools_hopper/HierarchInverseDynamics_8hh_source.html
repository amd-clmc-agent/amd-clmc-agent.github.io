76  template<int Max_Rows, typename Mat_Type, typename Vec_Type>
77  void appendRowsOfRank(
78  int rank_to_add,
79  const typename RtVectorX<Max_Rows>::i& ranks,
80  const Eigen::MatrixBase<Mat_Type>& subst_mat,
81  const Eigen::MatrixBase<Vec_Type>& subst_vec,
82  bool append_to_equalities,
83  int starting_column)
84  {
85  // count how many rows we add in this rank
86  int n_add_rows = 0;
87  for(int i=0; i<ranks.size(); ++i)
88  if(ranks[i] == rank_to_add)
89  ++n_add_rows;
90 
91  if(0 == n_add_rows) {
92  return; //nothing to be done
93  }
94  else
95  {
96  // extend the problem matrix by the amount of columns we need
97  int old_rows;
98  if(append_to_equalities)
99  {
100  old_rows = next_eq_cost_mat_.rows();
101  RtMatrixXUtils::conservativeResize(next_eq_cost_mat_,
102  old_rows+n_add_rows, next_eq_cost_mat_.cols());
103  RtVectorXUtils::conservativeResize(
104  next_eq_cost_vec_, old_rows+n_add_rows);
105  next_eq_cost_mat_.bottomRows(n_add_rows).setZero();
106  next_eq_cost_vec_.bottomRows(n_add_rows).setZero();
107  }
108  else
109  {
110  old_rows = next_ineq_cost_mat_.rows();
111  RtMatrixXUtils::conservativeResize(next_ineq_cost_mat_,
112  old_rows+n_add_rows, next_ineq_cost_mat_.cols());
113  RtVectorXUtils::conservativeResize(
114  next_ineq_cost_vec_, old_rows+n_add_rows);
115  next_ineq_cost_mat_.bottomRows(n_add_rows).setZero();
116  next_ineq_cost_vec_.bottomRows(n_add_rows).setZero();
117  }
118 
119  // add the rows from mat that correspond with the requested rank
120  int added_row=0;
121  for(int i=0;i <ranks.size(); ++i)
122  {
123  if(ranks[i] == rank_to_add)
124  {
125  if(append_to_equalities)
126  {
127  next_eq_cost_mat_.block(old_rows+added_row, starting_column, 1, subst_mat.cols()) = subst_mat.row(i);
128  next_eq_cost_vec_(old_rows+added_row) = subst_vec(i);
129  }
130  else
131  {
132  next_ineq_cost_mat_.block(old_rows+added_row, starting_column, 1, subst_mat.cols()) = subst_mat.row(i);
133  next_ineq_cost_vec_(old_rows+added_row) = subst_vec(i);
134  }
135  ++added_row;
136  }
137  }
138  }
139  }
