198  void ContactHelper::constructMomrateMap(
199  Eigen::MatrixBase<Mat> &mat, Eigen::MatrixBase<Vec> &vec) const {
200  RtVectorXUtils::setZero(vec, 6);
201  vec(2) = -kinematics_->robot_mass() * 9.81;
202 
203  for (int eff_i = 1; eff_i <= Robot::n_endeffs_; ++eff_i) {
204  mat.block(0, 6 * (eff_i - 1), 6, 6) =
205  getContactForceToMomentumRateMap(eff_i);
206  }
